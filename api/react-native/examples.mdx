## Overview

These examples use the TypeScript helper function from the [setup guide](/api/react-native/setup). All examples include proper typing and error handling.

## Identity

Get the current authenticated user's information.

```typescript
const me = await api<{ data: { id: string; email?: string } }>("/api/me");

if (!me.ok) {
  console.error("Failed to fetch user:", me.error);
} else {
  console.log("User ID:", me.data.id);
  console.log("Email:", me.data.email);
}
```

## Error Handling

The API helper uses discriminated unions for type-safe error handling. Always check the `ok` field before accessing data:

```typescript
const res = await api<{ data: unknown[] }>("/api/notes");

if (!res.ok) {
  // Handle error - TypeScript knows res.error exists
  console.log("API error:", res.error);
  
  // Show user-friendly message
  Alert.alert("Error", `Failed to load notes: ${res.error}`);
} else {
  // Success - TypeScript knows res.data exists
  console.log("Notes:", res.data);
  
  // Update your UI
  setNotes(res.data);
}
```

## Complete Example: Note Manager

Here's a complete example of a React Native component that manages notes:

```typescript
import React, { useState, useEffect } from 'react';
import { View, FlatList, Button, Alert } from 'react-native';

interface Note {
  id: string;
  content: string;
  title?: string;
}

function NoteManager() {
  const [notes, setNotes] = useState<Note[]>([]);
  const [loading, setLoading] = useState(true);

  // Load notes on mount
  useEffect(() => {
    loadNotes();
  }, []);

  const loadNotes = async () => {
    const res = await api<{ data: Note[] }>("/api/notes");
    
    if (!res.ok) {
      Alert.alert("Error", res.error);
    } else {
      setNotes(res.data);
    }
    
    setLoading(false);
  };

  const createNote = async () => {
    const res = await api<{ data: Note }>("/api/notes", {
      method: "POST",
      body: JSON.stringify({ content: "New note" }),
    });

    if (!res.ok) {
      Alert.alert("Error", res.error);
    } else {
      setNotes([...notes, res.data]);
    }
  };

  const deleteNote = async (noteId: string) => {
    const res = await api<{ data: unknown }>(
      `/api/notes/${noteId}/trash`,
      { method: "POST" }
    );

    if (!res.ok) {
      Alert.alert("Error", res.error);
    } else {
      setNotes(notes.filter(note => note.id !== noteId));
    }
  };

  return (
    <View>
      <Button title="Create Note" onPress={createNote} />
      <FlatList
        data={notes}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <NoteItem 
            note={item} 
            onDelete={() => deleteNote(item.id)}
          />
        )}
      />
    </View>
  );
}
```

<Note>
  Replace `<note-id>`, `<folder-id>`, and other placeholder values with actual IDs from your data.
</Note>
